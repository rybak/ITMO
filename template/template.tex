\begin{frame}[fragile]
   \frametitle{Вычисление предиката $Turn$}
   \framesubtitle{Числа с плавающей точкой}

   Двоичное число с плавающей точкой: $a = \pm 1.a_1a_2a_3\dots a_{p - 1} \times 2^\gamma, a_i \in \{0, 1\}.$
   
   \pause
   \begin{mydefinition}
      $ulp(a) = 2^{-p+1+\gamma}$
   \end{mydefinition}

   \pause
   $\mathit{D}$ --- множество всех чисел c плавающей точкой c операциями $\oplus, \ominus, \otimes$.
   
   \pause

   $\forall a, b \in \mathit{D},\ \varepsilon = 2^{-p}$:
   \begin{itemize}
      \item $a \oplus b = (a + b)(1 + \delta),\ |\delta| \leq \varepsilon$
      \item $a \ominus b = (a - b)(1 + \delta),\ |\delta| \leq \varepsilon$
      \item $a \otimes b = a b (1 + \delta),\ |\delta| \leq \varepsilon$
   \end{itemize}

   \pause
   \begin{statement}
      $\forall a, b, c \in \mathit{D}^2,\ \tilde{v} = (b_x \ominus a_x) \otimes (c_y \ominus a_y) \ominus (b_y \ominus a_y) \otimes (c_x \ominus a_x)$
      $\exists \tilde{\epsilon} \in \mathit{D}:
      \begin{cases}
         \tilde{v} > \tilde{\epsilon} \Rightarrow (b - a)  \times (c - a) > 0\\
         \tilde{v} < -\tilde{\epsilon} \Rightarrow (b - a) \times (c - a) < 0 
      \end{cases}$
   \end{statement}
\end{frame}
